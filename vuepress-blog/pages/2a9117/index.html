mysql备份

# 备份

LogFile=$BakDir/binlog.log
BinDir=/var/lib/mysql
Counter=`wc -l $BinFile |awk '{print $1}'`
for file in `cat $BinFile`
do
    base=`basename $file`
    echo $base
    #basename用于截取mysql-bin.00000*文件名，去掉./mysql-bin.000005前面的./
    NextNum=`expr $NextNum + 1`
    if [ $NextNum -gt $Counter ]
    then
        echo $base skip! >> $LogFile
    else
        dest=$BakDir/$base
        if test -e $dest
        #test -e用于检测目标文件是否存在，存在就写exist!到$LogFile去。
        then
            echo $base exist! >> $LogFile
        else
            cp $BinDir/$base $BakDir
            echo $base copying >> $LogFile
        fi
    fi
done
#sleep 5
dumpdate=$(date +%H%M%S)
filedate=$(date +%y%m%d)
mulu=/backup/sqlbackup/all/$filedate
if [ ! -d $mulu  ];then
 mkdir -p $mulu
fi
# 备份指定数据库中数据(此处假设数据库是mysql_backup_test)
mysql_user=""
mysql_password="!"
mysql_host="localhost"
mysql_port="20002"
mysql_charset="utf8"
source /etc/profile
docker exec  mysql mysqldump -h$mysql_host -P$mysql_port -u$mysql_user -p$mysql_password -B  insight_base -R  > ${mulu}/insight_base${filedate}.sql
#docker exec  mysql /usr/bin/mysqldump -h$mysql_host -P$mysql_port -u$mysql_user -p$mysql_password --events --quick --flush-logs --delete-master-logs --single-transaction -B  insight_base -R  > ${mulu}/insight_base${filedate}.sql
#删除第一行的警告信息 mysqldump: [Warning] Using a password on the command line interface can be insecure.
sed -i '1d'  ${mulu}/insight_base${filedate}.sql
echo `date +"%Y年%m月%d日 %H:%M:%S"` Bakup succ! >> $LogFile
