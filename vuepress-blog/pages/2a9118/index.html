golang项目docker部署

# 方式一

FROM golang:alpine as builder

WORKDIR /software/work/beego
COPY . .

RUN go env -w GO111MODULE=on \
    && go env -w GOPROXY=https://goproxy.cn,direct \
    && go env -w CGO_ENABLED=0 
FROM alpine:latest

LABEL MAINTAINER="1358338462@qq.com"

WORKDIR /software/work/beego

COPY --from=0 /software/work/beego/hello-beego ./
COPY --from=0 /software/work/beego/conf/app.conf ./conf/app.conf
COPY --from=0 /software/work/beego/static ./static/
COPY --from=0 /software/work/beego/views ./views/
#COPY --from=0 /go/src/github.com/flipped-aurora/gin-vue-admin/server/config.docker.yaml ./
EXPOSE 8080
ENTRYPOINT ./hello-beego
#bee pack -be GOOS=linux


FROM golang:alpine as builder

WORKDIR /software/work/hello-beego
COPY . .

RUN go env -w GO111MODULE=on \
    && go env -w GOPROXY=https://goproxy.cn,direct \
    && go env -w CGO_ENABLED=0 \
    && go env \
    && go mod tidy \
    && go build -o server .

FROM alpine:latest
ENV TZ Asia/Shanghai
LABEL MAINTAINER="1358338462@qq.com"

WORKDIR /software/work/hello-beego

COPY --from=0 /software/work/hello-beego/server ./
COPY --from=0 /software/work/hello-beego/conf/app.conf ./conf/app.conf
COPY --from=0 /software/work/hello-beego/static ./static/
COPY --from=0 /software/work/hello-beego/views ./views/
#COPY --from=0 /go/src/github.com/flipped-aurora/gin-vue-admin/server/config.docker.yaml ./
EXPOSE 8080
ENTRYPOINT ./server
